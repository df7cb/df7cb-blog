        array_accum(SUBSTRING(source FROM 1 FOR 1)) AS first,
        SUBSTRING(source FROM 2) AS rest
FROM
        (SELECT DISTINCT source FROM source) AS source
GROUP BY rest
HAVING
        COUNT(SUBSTRING(source FROM 1 FOR 1)) >= 4
ORDER BY rest;
